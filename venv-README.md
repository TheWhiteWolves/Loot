# Virtual Environment Setup
# To activate the virtual environment in PowerShell:
# .\venv\Scripts\Activate.ps1

# To run the bot:
# .\venv\Scripts\Activate.ps1 && python bot.py

# Or create an alias in PowerShell:
# function Activate-Loot { .\venv\Scripts\Activate.ps1 }
# Then use: Activate-Loot && python bot.py
